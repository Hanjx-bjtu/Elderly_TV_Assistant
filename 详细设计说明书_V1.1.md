### 3.1 启动模块（SplashActivity）

#### 3.1.1 功能描述
- 展示启动画面（2秒）
- 初始化本地数据库（首次启动时加载预设频道数据）
- 判断是否首次启动，决定跳转引导页或主页

#### 3.1.2 类设计（Kotlin）

```kotlin
class SplashActivity : AppCompatActivity() {
    
    private lateinit var mHandler: Handler
    private lateinit var mDatabase: AppDatabase
    private lateinit var mPrefsHelper: SharedPrefsHelper
    
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_splash)
        
        mHandler = Handler(Looper.getMainLooper())
        mPrefsHelper = SharedPrefsHelper(this)
        
        initDatabase()
    }
    
    private fun initDatabase() {
        // 使用协程进行数据库初始化
        lifecycleScope.launch(Dispatchers.IO) {
            mDatabase = AppDatabase.getInstance(this@SplashActivity)
            
            // 检查是否需要初始化预设数据
            if (mDatabase.channelDao().getChannelCount() == 0) {
                loadPresetChannels()
            }
            
            // 跳转到下一页面
            withContext(Dispatchers.Main) {
                checkFirstLaunch()
            }
        }
    }
    
    private suspend fun loadPresetChannels() {
        val channels = ChannelDataLoader.loadFromAssets(this)
        mDatabase.channelDao().insertAllChannels(channels)
    }
    
    private fun checkFirstLaunch() {
        val isFirstLaunch = mPrefsHelper.isFirstLaunch()
        val intent = if (isFirstLaunch) {
            Intent(this, GuideActivity::class.java)
        } else {
            Intent(this, MainActivity::class.java)
        }
        
        mHandler.postDelayed({
            startActivity(intent)
            finish()
        }, 2000)
    }
}
```

```kotlin
class MainActivity : AppCompatActivity() {
    
    // UI组件
    private lateinit var mDrawerLayout: DrawerLayout
    private lateinit var mCurrentChannelText: TextView
    private lateinit var mVoiceBtn: ImageButton
    private lateinit var mTimerBtn: ImageButton
    private lateinit var mFavoriteBtn: ImageButton
    private lateinit var mGestureLayout: GestureRelativeLayout
    
    // 管理器
    private lateinit var mPlayerManager: PlayerManager
    private lateinit var mVoiceManager: VoiceManager
    private lateinit var mTTSManager: TTSManager
    private lateinit var mTimerManager: TimerManager
    
    // ViewModel
    private lateinit var mChannelViewModel: ChannelViewModel
    private lateinit var mVoiceViewModel: VoiceViewModel
    private lateinit var mTimerViewModel: TimerViewModel
    
    // 数据
    private lateinit var mChannelList: List<Channel>
    private var mCurrentChannelIndex = 0
    private var mCurrentChannel: Channel? = null
    
    // 设置
    private lateinit var mPrefsHelper: SharedPrefsHelper
    
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)
        
        initViews()
        initManagers()
        initViewModels()
        setupObservers()
        setupDrawer()
        setupGesture()
        setupBottomButtons()
        
        // 处理定时关闭退出
        if (intent?.getBooleanExtra("TIMER_OFF", false) == true) {
            finish()
        }
    }
    
    private fun initViews() {
        mDrawerLayout = findViewById(R.id.drawer_layout)
        mCurrentChannelText = findViewById(R.id.tv_current_channel)
        mVoiceBtn = findViewById(R.id.btn_voice)
        mTimerBtn = findViewById(R.id.btn_timer)
        mFavoriteBtn = findViewById(R.id.btn_favorite)
        mGestureLayout = findViewById(R.id.gesture_layout)
    }
    
    private fun initManagers() {
        val videoContainer = findViewById<FrameLayout>(R.id.video_container)
        mPlayerManager = PlayerManager(this, videoContainer)
        mPlayerManager.init(PlayerManager.PlayerType.WEB_VIEW)
        
        mTTSManager = TTSManager(this)
        mTTSManager.init()
        
        mTimerManager = TimerManager(this)
        
        mPrefsHelper = SharedPrefsHelper(this)
    }
    
    private fun initViewModels() {
        mChannelViewModel = ViewModelProvider(this)[ChannelViewModel::class.java]
        mVoiceViewModel = ViewModelProvider(this)[VoiceViewModel::class.java]
        mTimerViewModel = ViewModelProvider(this)[TimerViewModel::class.java]
    }
    
    private fun setupObservers() {
        // 观察频道列表变化
        mChannelViewModel.allChannels.observe(this) { channels ->
            mChannelList = channels
            // 设置语音关键词
            mVoiceManager.updateKeywords(channels)
            // 启动默认频道
            val defaultChannelId = mPrefsHelper.defaultChannelId
            val defaultChannel = channels.find { it.id == defaultChannelId } ?: channels.firstOrNull()
            defaultChannel?.let { switchToChannel(it) }
        }
        
        // 观察收藏状态变化
        mChannelViewModel.isFavorite.observe(this) { isFavorite ->
            mFavoriteBtn.isSelected = isFavorite
        }
        
        // 观察语音识别结果
        mVoiceViewModel.voiceResult.observe(this) { channelName ->
            channelName?.let {
                val targetChannel = mChannelList.find { channel ->
                    channel.name.contains(it, ignoreCase = true) ||
                    channel.keywords.any { keyword -> keyword.contains(it, ignoreCase = true) }
                }
                targetChannel?.let { channel ->
                    switchToChannel(channel)
                    if (mPrefsHelper.isTTSEnabled) {
                        mTTSManager.speakChannel(channel.name)
                    }
                }
            }
        }
        
        // 观察定时器状态
        mTimerViewModel.timerRemaining.observe(this) { remaining ->
            if (remaining > 0) {
                // 更新UI显示倒计时（可扩展）
            }
        }
    }
    
    private fun setupGesture() {
        mGestureLayout.setOnRegionClickListener(object : GestureRelativeLayout.OnRegionClickListener {
            override fun onLeftClick() {
                switchToPrevChannel()
                vibrate()
            }
            
            override fun onRightClick() {
                switchToNextChannel()
                vibrate()
            }
            
            override fun onMiddleClick() {
                // 中间区域点击可扩展显示/隐藏控制栏
                toggleControlBar()
            }
        })
    }
    
    private fun setupBottomButtons() {
        mVoiceBtn.setOnClickListener {
            startVoiceRecognition()
        }
        
        mTimerBtn.setOnClickListener {
            showTimerDialog()
        }
        
        mFavoriteBtn.setOnClickListener {
            toggleFavorite()
        }
    }
    
    fun switchToPrevChannel() {
        if (mChannelList.isEmpty()) return
        mCurrentChannelIndex = (mCurrentChannelIndex - 1 + mChannelList.size) % mChannelList.size
        switchToChannel(mChannelList[mCurrentChannelIndex])
    }
    
    fun switchToNextChannel() {
        if (mChannelList.isEmpty()) return
        mCurrentChannelIndex = (mCurrentChannelIndex + 1) % mChannelList.size
        switchToChannel(mChannelList[mCurrentChannelIndex])
    }
    
    fun switchToChannel(channel: Channel) {
        mCurrentChannel = channel
        mCurrentChannelIndex = mChannelList.indexOf(channel)
        mCurrentChannelText.text = channel.displayName ?: channel.name
        
        // 更新收藏按钮状态
        mChannelViewModel.checkFavorite(channel.id)
        
        // 播放视频
        mPlayerManager.play(channel.url)
        
        // 语音播报
        if (mPrefsHelper.isTTSEnabled) {
            mTTSManager.speakChannel(channel.name)
        }
    }
    
    private fun startVoiceRecognition() {
        if (checkPermission(Manifest.permission.RECORD_AUDIO)) {
            mVoiceManager.startListening()
            // 显示语音识别动画
            showVoiceAnimation()
        } else {
            requestPermission(Manifest.permission.RECORD_AUDIO, PERMISSION_VOICE_CODE)
        }
    }
    
    private fun showTimerDialog() {
        val minutes = arrayOf("15分钟", "30分钟", "60分钟", "90分钟", "取消定时")
        AlertDialog.Builder(this)
            .setTitle("定时关闭")
            .setItems(minutes) { _, which ->
                when (which) {
                    0 -> setTimer(15)
                    1 -> setTimer(30)
                    2 -> setTimer(60)
                    3 -> setTimer(90)
                    4 -> cancelTimer()
                }
            }
            .show()
    }
    
    private fun setTimer(minutes: Int) {
        mTimerManager.setTimer(minutes)
        Toast.makeText(this, "已设置${minutes}分钟后自动关闭", Toast.LENGTH_SHORT).show()
    }
    
    private fun cancelTimer() {
        mTimerManager.cancelTimer()
        Toast.makeText(this, "已取消定时关闭", Toast.LENGTH_SHORT).show()
    }
    
    private fun toggleFavorite() {
        mCurrentChannel?.let { channel ->
            mChannelViewModel.toggleFavorite(channel.id)
        }
    }
    
    private fun vibrate() {
        if (mPrefsHelper.isVibrateEnabled) {
            val vibrator = getSystemService(Context.VIBRATOR_SERVICE) as Vibrator
            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
                vibrator.vibrate(VibrationEffect.createOneShot(50, VibrationEffect.DEFAULT_AMPLITUDE))
            } else {
                vibrator.vibrate(50)
            }
        }
    }
    
    override fun onBackPressed() {
        AlertDialog.Builder(this)
            .setTitle("退出")
            .setMessage("确定要退出长辈电视助手吗？")
            .setPositiveButton("确定") { _, _ ->
                finish()
            }
            .setNegativeButton("取消", null)
            .show()
    }
    
    override fun onDestroy() {
        super.onDestroy()
        mPlayerManager.release()
        mVoiceManager.destroy()
        mTTSManager.shutdown()
        mTimerManager.cancelTimer()
    }
    
    companion object {
        private const val PERMISSION_VOICE_CODE = 100
    }
}
```

```kotlin
// GestureRelativeLayout.kt
class GestureRelativeLayout @JvmOverloads constructor(
    context: Context,
    attrs: AttributeSet? = null,
    defStyleAttr: Int = 0
) : RelativeLayout(context, attrs, defStyleAttr) {
    
    companion object {
        private const val LEFT_REGION = 0      // 左侧区域
        private const val RIGHT_REGION = 1     // 右侧区域
        private const val MIDDLE_REGION = 2    // 中间区域
    }
    
    private var listener: OnRegionClickListener? = null
    
    override fun onTouchEvent(event: MotionEvent): Boolean {
        when (event.action) {
            MotionEvent.ACTION_UP -> {
                val x = event.x
                val width = width.toFloat()
                
                when {
                    x < width / 3 -> listener?.onLeftClick()
                    x > width * 2 / 3 -> listener?.onRightClick()
                    else -> listener?.onMiddleClick()
                }
            }
        }
        return true
    }
    
    fun setOnRegionClickListener(listener: OnRegionClickListener) {
        this.listener = listener
    }
    
    interface OnRegionClickListener {
        fun onLeftClick()
        fun onRightClick()
        fun onMiddleClick()
    }
}
```

### 3.3 视频播放模块（PlayerManager）

#### 3.3.1 功能描述
- 管理视频播放器的创建、销毁
- 支持两种播放方案：WebView（主方案）+ IjkPlayer（备选）
- 提供播放、停止、切换视频源接口

#### 3.3.2 类设计（Kotlin）

```kotlin
class PlayerManager(
    private val context: Context,
    private val container: FrameLayout
) {
    
    enum class PlayerType {
        WEB_VIEW,   // WebView方案
        IJK_PLAYER  // IjkPlayer方案（备选）
    }
    
    private var playerType: PlayerType = PlayerType.WEB_VIEW
    private var webView: WebView? = null
    private var ijkPlayer: IjkMediaPlayer? = null
    private var currentUrl: String = ""
    private var isPlaying = false
    
    fun init(type: PlayerType) {
        playerType = type
        when (type) {
            PlayerType.WEB_VIEW -> initWebView()
            PlayerType.IJK_PLAYER -> initIjkPlayer()
        }
    }
    
    private fun initWebView() {
        webView = WebView(context).apply {
            layoutParams = FrameLayout.LayoutParams(
                FrameLayout.LayoutParams.MATCH_PARENT,
                FrameLayout.LayoutParams.MATCH_PARENT
            )
            
            settings.apply {
                javaScriptEnabled = true
                domStorageEnabled = true
                loadWithOverviewMode = true
                useWideViewPort = true
                mediaPlaybackRequiresUserGesture = false
                mixedContentMode = WebSettings.MIXED_CONTENT_ALWAYS_ALLOW
                cacheMode = WebSettings.LOAD_DEFAULT
            }
            
            webChromeClient = WebChromeClient()
            webViewClient = object : WebViewClient() {
                override fun onPageFinished(view: WebView?, url: String?) {
                    super.onPageFinished(view, url)
                    // 页面加载完成后尝试自动播放
                    evaluateJavascript("document.querySelector('video')?.play()", null)
                }
                
                override fun onReceivedError(
                    view: WebView?,
                    request: WebResourceRequest?,
                    error: WebResourceError?
                ) {
                    // 错误处理回调
                }
            }
        }
        
        container.removeAllViews()
        container.addView(webView)
    }
    
    private fun initIjkPlayer() {
        // IjkPlayer初始化逻辑（备用方案）
        IjkMediaPlayer.loadLibrariesOnce(null)
        IjkMediaPlayer.native_profileBegin("libijkplayer.so")
        
        ijkPlayer = IjkMediaPlayer().apply {
            setOption(IjkMediaPlayer.OPT_CATEGORY_PLAYER, "opensles", 0)
            setOption(IjkMediaPlayer.OPT_CATEGORY_PLAYER, "overlay-format", IjkMediaPlayer.SDL_FCC_RV32)
            setOption(IjkMediaPlayer.OPT_CATEGORY_PLAYER, "framedrop", 1)
            setOption(IjkMediaPlayer.OPT_CATEGORY_PLAYER, "start-on-prepared", 1)
            setOption(IjkMediaPlayer.OPT_CATEGORY_FORMAT, "http-detect-range-support", 1)
        }
        
        val surfaceView = SurfaceView(context)
        container.addView(surfaceView)
        ijkPlayer?.setDisplay(surfaceView.holder)
    }
    
    fun play(url: String) {
        currentUrl = url
        when (playerType) {
            PlayerType.WEB_VIEW -> {
                webView?.loadUrl(url)
            }
            PlayerType.IJK_PLAYER -> {
                try {
                    ijkPlayer?.apply {
                        stop()
                        reset()
                        setDataSource(url)
                        prepareAsync()
                    }
                } catch (e: Exception) {
                    e.printStackTrace()
                }
            }
        }
        isPlaying = true
    }
    
    fun stop() {
        when (playerType) {
            PlayerType.WEB_VIEW -> {
                webView?.loadUrl("about:blank")
            }
            PlayerType.IJK_PLAYER -> {
                ijkPlayer?.stop()
            }
        }
        isPlaying = false
    }
    
    fun release() {
        stop()
        when (playerType) {
            PlayerType.WEB_VIEW -> {
                webView?.destroy()
                webView = null
            }
            PlayerType.IJK_PLAYER -> {
                ijkPlayer?.release()
                ijkPlayer = null
            }
        }
        container.removeAllViews()
    }
    
    fun setVolume(volume: Float) {
        ijkPlayer?.setVolume(volume, volume)
    }
    
    fun isPlaying(): Boolean = isPlaying
}
```

### 3.4 语音模块（VoiceManager + TTSManager）

#### 3.4.1 功能描述

- 语音合成：切换频道时自动播报当前频道名称

#### 3.4.2 VoiceManager类设计（Kotlin）

```kotlin
class VoiceManager(
    private val context: Context,
    private val callback: VoiceCallback
) {
    
    interface VoiceCallback {
        fun onListeningStart()
        fun onResult(channelName: String?)
        fun onError(error: String)
    }
    
    private var speechRecognizer: SpeechRecognizer? = null
    private var channelKeywords: Map<String, String> = emptyMap()  // keyword -> channelName
    
    fun init() {
        speechRecognizer = SpeechRecognizer.createSpeechRecognizer(context)
        speechRecognizer?.setRecognitionListener(object : RecognitionListener {
            override fun onReadyForSpeech(params: Bundle?) {
                callback.onListeningStart()
            }
            
            override fun onBeginningOfSpeech() {}
            
            override fun onRmsChanged(rmsDB: Float) {}
            
            override fun onBufferReceived(buffer: ByteArray?) {}
            
            override fun onEndOfSpeech() {}
            
            override fun onError(error: Int) {
                callback.onError("识别失败: $error")
            }
            
            override fun onResults(results: Bundle?) {
                val matches = results?.getStringArrayList(SpeechRecognizer.RESULTS_RECOGNITION)
                val spokenText = matches?.firstOrNull() ?: ""
                val matchedChannel = matchChannel(spokenText)
                callback.onResult(matchedChannel)
            }
            
            override fun onPartialResults(partialResults: Bundle?) {}
            
            override fun onEvent(eventType: Int, params: Bundle?) {}
        })
    }
    
    fun updateKeywords(channels: List<Channel>) {
        channelKeywords = buildMap {
            channels.forEach { channel ->
                channel.keywords.forEach { keyword ->
                    put(keyword.lowercase(), channel.name)
                }
            }
        }
    }
    
    private fun matchChannel(spokenText: String): String? {
        val lowerText = spokenText.lowercase()
        for ((keyword, channelName) in channelKeywords) {
            if (lowerText.contains(keyword)) {
                return channelName
            }
        }
        return null
    }
    
    fun startListening() {
        val intent = Intent(RecognizerIntent.ACTION_RECOGNIZE_SPEECH).apply {
            putExtra(RecognizerIntent.EXTRA_LANGUAGE_MODEL, RecognizerIntent.LANGUAGE_MODEL_FREE_FORM)
            putExtra(RecognizerIntent.EXTRA_LANGUAGE, Locale.CHINESE.toString())
            putExtra(RecognizerIntent.EXTRA_PROMPT, "请说出频道名称")
        }
        speechRecognizer?.startListening(intent)
    }
    
    fun stopListening() {
        speechRecognizer?.stopListening()
    }
    
    fun destroy() {
        speechRecognizer?.destroy()
        speechRecognizer = null
    }
}
```

```kotlin
    }

        tts?.stop()
    }

        tts?.stop()
        tts?.shutdown()

    }
```

#### 3.5.2 类设计（Kotlin）

```kotlin
// TimerManager.kt
class TimerManager(private val context: Context) {
    
    private val alarmManager = context.getSystemService(Context.ALARM_SERVICE) as AlarmManager
    private var pendingIntent: PendingIntent? = null
    
    fun setTimer(minutes: Int) {
        cancelTimer()
        
        val triggerTime = System.currentTimeMillis() + (minutes * 60 * 1000L)
        val intent = Intent(context, TimerReceiver::class.java)
        
        pendingIntent = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
            PendingIntent.getBroadcast(context, 0, intent, 
                PendingIntent.FLAG_UPDATE_CURRENT or PendingIntent.FLAG_IMMUTABLE)
        } else {
            PendingIntent.getBroadcast(context, 0, intent, PendingIntent.FLAG_UPDATE_CURRENT)
        }
        
        alarmManager.setExact(AlarmManager.RTC_WAKEUP, triggerTime, pendingIntent)
    }
    
    fun cancelTimer() {
        pendingIntent?.let {
            alarmManager.cancel(it)
            pendingIntent = null
        }
    }
    
    fun isTimerSet(): Boolean = pendingIntent != null
}

// TimerReceiver.kt
class TimerReceiver : BroadcastReceiver() {
    
    override fun onReceive(context: Context, intent: Intent) {
        // 发送通知
        sendNotification(context)
        
        // 关闭MainActivity
        val closeIntent = Intent(context, MainActivity::class.java).apply {
            flags = Intent.FLAG_ACTIVITY_NEW_TASK
            putExtra("TIMER_OFF", true)
        }
        context.startActivity(closeIntent)
    }
    
    private fun sendNotification(context: Context) {
        val notificationManager = context.getSystemService(Context.NOTIFICATION_SERVICE) as NotificationManager
        
        // 创建通知渠道（Android 8.0+）
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            val channel = NotificationChannel(
                "timer_channel",
                "定时提醒",
                NotificationManager.IMPORTANCE_HIGH
            ).apply {
                description = "定时关闭提醒"
            }
            notificationManager.createNotificationChannel(channel)
        }
        
        val notification = NotificationCompat.Builder(context, "timer_channel")
            .setContentTitle("长辈电视助手")
            .setContentText("播放时间已到")
            .setSmallIcon(android.R.drawable.ic_dialog_info)
            .setAutoCancel(true)
            .build()
        
        notificationManager.notify(1, notification)
    }
}
```

```kotlin
        vibrateSwitch?.setOnPreferenceChangeListener { _, newValue ->
            prefsHelper.setVibrateEnabled(newValue as Boolean)
            true
        }
        
        // 默认频道设置

            .setSingleChoiceItems(sizes, current) { _, which ->
                prefsHelper.setFontSize(which + 1)
                // 通知Activity更新字体
                sendBroadcast(Intent("FONT_SIZE_CHANGED"))
                recreate()
            }
            .setPositiveButton("确定", null)
            .show()
```

## 6. 异常处理设计

### 6.1 异常处理总览

| 异常场景 | 处理方式 | 用户提示 |
|---------|---------|---------|
| 网络不可用 | 显示提示，播放本地缓存提示页面 | “请检查网络连接” |
| 视频加载失败 | 自动重试1次，失败后提示 | “播放失败，请稍后重试” |
| 语音识别失败 | 静默处理，不中断当前播放 | 无提示（避免困扰） |
| 定时关闭冲突 | 取消旧定时，设置新定时 | “已设置X分钟后自动关闭” |
| 数据库初始化失败 | 使用默认频道列表（硬编码） | “加载失败，使用默认频道” |
| 录音权限拒绝 | 弹窗引导用户授权 | “需要录音权限才能使用语音功能” |

## 7. 性能优化设计

| 优化点 | 实现方式 |
|-------|---------|
| WebView复用 | 切换频道时复用WebView实例，仅改变URL |
| 数据库异步操作 | Room默认异步查询，LiveData自动感知生命周期 |
| 图片加载 | 使用Glide缓存频道图标 |
| 语音识别资源释放 | 页面销毁时及时释放SpeechRecognizer |
| 后台资源 | TimerManager不使用Service，仅用AlarmManager降低功耗 |
| 启动优化 | SplashActivity中使用协程异步初始化数据库 |
| 内存管理 | 及时释放WebView和播放器资源 |
